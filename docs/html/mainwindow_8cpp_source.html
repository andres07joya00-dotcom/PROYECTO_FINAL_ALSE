263void MainWindow::onEdit()
264{
265 QModelIndexList sel = tableView->selectionModel()->selectedRows();
266 if (sel.isEmpty()) {
267 QMessageBox::information(this, "Editar", "Por favor, selecciona una fila primero.");
268 return;
269 }
270
271 // Mapeo del índice del proxy al índice del modelo fuente para obtener el ID real
272 QModelIndex idx = sel.first();
273 QModelIndex src = proxy->mapToSource(idx);
274 int id = model->data(model->index(src.row(), 0)).toInt();
275
276 InventoryItem it = manager.getItemById(id);
277
278 AddDialog *dlg = new AddDialog();
279 dlg->setWindowModality(Qt::ApplicationModal);
280
281 // Precargar datos en los widgets del diálogo usando findChild
282 dlg->findChild<QLineEdit*>("nameEdit")->setText(it.nombre);
283 dlg->findChild<QLineEdit*>("typeEdit")->setText(it.tipo);
284 dlg->findChild<QSpinBox*>("qtySpin")->setValue(it.cantidad);
285 dlg->findChild<QLineEdit*>("locEdit")->setText(it.ubicacion);
286 dlg->findChild<QDateEdit*>("dateEdit")->setDate(QDate::fromString(it.fechaAdquisicion, "yyyy-MM-dd"));
287
288 dlg->show();
289
290 // Lógica al aceptar el diálogo
291 connect(dlg, &AddDialog::accepted, this, [this, dlg, id]() {
292
293 if (dlg->getName().isEmpty() || dlg->getType().isEmpty()) {
294 QMessageBox::warning(this, "Validación", "Nombre y tipo son campos obligatorios.");
295 return;
296 }
297
298 if (!manager.updateItem(
299 id,
300 dlg->getName(),
301 dlg->getType(),
302 dlg->getQuantity(),
303 dlg->getLocation(),
304 dlg->getDate().toString("yyyy-MM-dd")))
305 {
306 QMessageBox::critical(this, "Error", "Fallo al actualizar el componente en la base de datos.");
307 } else {
308 refreshModel();
309 }
310
311 dlg->close();
312 dlg->deleteLater();
313 });
314
315 connect(dlg, &AddDialog::cancelled, dlg, &AddDialog::close);
316}

494void MainWindow::onLowStock()
495{
496 QStringList lowStockItems;
497
498 for (int r = 0; r < proxy->rowCount(); r++) {
499 // Índice de la columna 'Cantidad' (columna 3)
500 QModelIndex idxQty = proxy->index(r, 3);
501 int qty = idxQty.data().toInt();
502
503 if (qty < lowStockThreshold) {
504 // Pintar toda la fila (iterando columnas)
505 for (int c = 0; c < proxy->columnCount(); c++) {
506 tableView->model()->setData(
507 proxy->index(r, c),
508 QColor(255, 200, 200), // Rojo claro
509 Qt::BackgroundRole
510 );
511 }
512
513 // Agregar nombre a la lista de alerta
514 QModelIndex idxName = proxy->index(r, 1);
515 lowStockItems << idxName.data().toString();
516 }
517 }
518
519 if (!lowStockItems.isEmpty()) {
520 QMessageBox::warning(this, "Alerta de Stock Bajo",
521 "Los siguientes ítems están por debajo del mínimo:\n\n" + lowStockItems.join("\n"));
522 } else {
523 QMessageBox::information(this, "Stock Saludable", "No hay elementos con stock bajo.");
524 }
525}
